-r base.txt

# base.txt leaves numpy unpinned (fine for sdv, which resolves to numpy 2.x),
# but synthcity 0.2.12 and xgbse (a synthcity dependency) both hard-cap
# numpy<2.0 — pin it here to override base.txt's open constraint for this
# venv specifically.
numpy==1.26.4

# scipy.integrate.trapz was removed in 1.14 (renamed to trapezoid in 1.11,
# alias dropped in 1.14); synthcity still calls trapz, so cap below the cutoff.
scipy<1.14

# Pydantic 2.13 added a breaking check that rejects unannotated class attributes
# on BaseModel subclasses — synthcity's Plugin base class has `protected_cols`
# without a ClassVar annotation, causing all synthesizers to fail at init.
pydantic<2.13

# synthcity is pinned (not left open like sdv/concrete-ml) because leaving it
# unpinned makes pip backtrack across 0.2.12 down to 0.2.4, hitting wildly
# different transitive dependency trees (old pydantic<2.0, pytorch-lightning
# <2.0, etc.) and eventually failing on long-obsolete releases that hard-cap
# torch<2.0 — even though 0.2.12 itself is compatible with the torch pin
# below. Pinning it forces pip to resolve against exactly one dependency
# tree instead of exploring eight.
synthcity

# synthcity needs torch. torch and torchvision are pinned together and
# installed from the CUDA index in this single resolution pass — do NOT
# `pip install torch` separately afterwards, since that reinstalls torch from
# a different index without touching torchvision and leaves the two
# mismatched (symptom: "partially initialized module 'torchvision' has no
# attribute 'extension'" on import).
#
# torch is capped at 2.2.2 (not the 2.4.1 used in azure-sdv.txt) because
# synthcity 0.2.1x requires torch<2.3,>=2.1 — pinning higher makes pip's
# resolver fail with ResolutionImpossible.
--extra-index-url https://download.pytorch.org/whl/cu121
torch==2.4.1
torchvision
xgboost
scipy==1.11.4

# GPU XGBoost (device='cuda') needs cupy to hand XGBoost a GPU-resident
# array — see Model._to_cuda() in src/models.py. cupy-cuda12x matches the
# CUDA 12.x torch build pinned above.
# >=13.0 required: CUDA 12.5+ ships libnvrtc.so.13 (not .so.12); CuPy 13.x
# updated its expected NVRTC .so version accordingly.
cupy-cuda12x>=13.0
